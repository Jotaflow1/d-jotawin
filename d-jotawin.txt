js/app.js


// carrega lista de jogos
fetch('js/games.json')
  .then(r => r.json())
  .then(list => {
    const container = document.getElementById('gameList');
    list.forEach(g => {
      const card = document.createElement('div');
      card.className = 'game-card';
      card.innerHTML = `
        <img src="${g.thumb}" alt="${g.name}" />
        <h3>${g.name}</h3>
        <button onclick="play('${g.id}')">Jogar</button>
      `;
      container.appendChild(card);
    });
  });

function play(id) {
  // redireciona para o jogo (iframe ou abre em nova aba)
  window.open(`https://games.d-jota-win.com/play/${id}`, '_blank');
}

    let balance = 12750.35;
    document.getElementById('balance').innerText = balance.toLocaleString('pt-BR', {style:'currency', currency:'BRL'});

    document.getElementById('cashForm').addEventListener('submit', e => {
      e.preventDefault();
      const val = parseFloat(document.getElementById('amount').value);
      if (val > balance) return alert('Saldo insuficiente');
      alert(`Saque de ${val.toLocaleString('pt-BR', {style:'currency', currency:'BRL'})} solicitado para ${document.getElementById('pix').value}`);
